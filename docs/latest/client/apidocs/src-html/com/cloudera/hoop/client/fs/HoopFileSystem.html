065      /**
066       * Convenience method that creates a <code>HttpURLConnection</code> for the Hoop file system operations.
067       * <p/>
068       * This methods performs and injects any needed authentication credentials.
069       *
070       * @param method the HTTP method.
071       * @param params the query string parameters.
072       * @param path the file path
073       * @return a <code>HttpURLConnection</code> for the Hoop server, authenticated and ready to use for
074       * the specified path and file system operation.
075       * @throws IOException
076       */
077      private HttpURLConnection getConnection(String method, Map<String, String> params, Path path) throws IOException {
078        params.put("doas", doAs);
079        Class<? extends Authenticator> klass =
080          getConf().getClass("hoop.authenticator.class", HoopKerberosAuthenticator.class, Authenticator.class);
081        Authenticator authenticator = ReflectionUtils.newInstance(klass, getConf());
082        try {
083          StringBuilder sb = new StringBuilder();
084          String separator = "?";
085          for (Map.Entry<String, String> entry : params.entrySet()) {
086            sb.append(separator).append(entry.getKey()).append("=").append(URLEncoder.encode(entry.getValue(),
087                                                                                             "UTF8"));
088            separator = "&";
089          }
090          path = makeQualified(path);
091          URL url = new URL(path + sb.toString());
092          HttpURLConnection conn = new AuthenticatedURL(authenticator).openConnection(url, authToken);
093          conn.setRequestMethod(method);
094          if (method.equals("POST") || method.equals("PUT")) {
095            conn.setDoOutput(true);
096          }
097          return conn;
098        }
099        catch (Exception ex) {
100          throw new IOException(ex);
101        }
102      }

104      /**
105       * Convenience method that JSON Parses the <code>InputStream</code> of a <code>HttpURLConnection</code>.
106       *
107       * @param conn the <code>HttpURLConnection</code>.
108       * @return the parsed JSON object.
109       * @throws IOException thrown if the <code>InputStream</code> could not be JSON parsed.
110       */
111      private static Object jsonParse(HttpURLConnection conn) throws IOException {
112        try {
113          JSONParser parser = new JSONParser();
114          return parser.parse(new InputStreamReader(conn.getInputStream()));
115        }
116        catch (ParseException ex) {
117          throw new IOException("JSON parser error, " + ex.getMessage(), ex);
118        }
119      }

121      /**
122       * Validates the status of an <code>HttpURLConnection</code> against an expected HTTP
123       * status code. If the current status code is not the expected one it throws an exception
124       * with a detail message using Server side error messages if available.
125       *
126       * @param conn the <code>HttpURLConnection</code>.
127       * @param expected the expected HTTP status code.
128       * @throws IOException thrown if the current status code does not match the expected one.
129       */
130      private static void validateResponse(HttpURLConnection conn, int expected) throws IOException {
131        int status = conn.getResponseCode();
132        if (status != expected) {
133          try {
134            JSONObject json = (JSONObject) jsonParse(conn);
135            throw new IOException(MessageFormat.format("HTTP status [{0}], {1} - {2}", json.get("status"),
136                                                       json.get("reason"), json.get("message")));
137          }
138          catch (IOException ex) {
139            if (ex.getCause() instanceof IOException) {
140              throw (IOException) ex.getCause();
141            }
142            throw new IOException(MessageFormat.format("HTTP status [{0}], {1}", status, conn.getResponseMessage()));
143          }
144        }
145      }

147      /**
148       * Called after a new FileSystem instance is constructed.
149       *
150       * @param name a uri whose authority section names the host, port, etc. for this FileSystem
151       * @param conf the configuration
152       */
153      @Override
154      public void initialize(URI name, Configuration conf) throws IOException {
155        UserGroupInformation ugi = UserGroupInformation.getCurrentUser();
156        doAs = ugi.getUserName();
157        super.initialize(name, conf);
158        try {
159          uri = new URI(name.getScheme() + "://" + name.getHost() + ":" + name.getPort());
160        }
161        catch (URISyntaxException ex) {
162          throw new IOException(ex);
163        }
164      }

219      /**
220       * Opens an FSDataInputStream at the indicated Path.
221       * </p>
222       * IMPORTANT: the returned <code><FSDataInputStream/code> does not support the
223       * <code>PositionReadable</code> and <code>Seekable</code> methods.
224       * 
225       * @param f the file name to open
226       * @param bufferSize the size of the buffer to be used.
227       */
228      @Override
229      public FSDataInputStream open(Path f, int bufferSize) throws IOException {
230        Map<String, String> params = new HashMap<String, String>();
231        HttpURLConnection conn = getConnection("GET", params, f);
232        validateResponse(conn, HttpURLConnection.HTTP_OK);
233        return new FSDataInputStream(new HoopFSDataInputStream(conn.getInputStream(), bufferSize));
234      }

236      /**
237       * Hoop subclass of the <code>FSDataOutputStream</code>.
238       * <p/>
239       * This implementation closes the underlying HTTP connection validating the Http connection status
240       * at closing time.
241       */
242      private static class HoopFSDataOutputStream extends FSDataOutputStream {
243        private HttpURLConnection conn;
244        private int closeStatus;
245    
246        public HoopFSDataOutputStream(HttpURLConnection conn, OutputStream out, int closeStatus, Statistics stats)
247          throws IOException {
248          super(out, stats);
249          this.conn = conn;
250          this.closeStatus = closeStatus;
251        }
252    
253        @Override
254        public void close() throws IOException {
255          try {
256            super.close();
257          }
258          finally {
259            validateResponse(conn, closeStatus);
260          }
261        }
262    
263      }

265      /**
266       * Converts a <code>FsPermission</code> to a Unix string symbolic representation (ie: '-rwxr--r--')
267       * @param p the permission.
268       * @return the Unix string symbolic reprentation.
269       */
270      private String permissionToString(FsPermission p) {
271        return (p == null) ? "default" : "-" + p.getUserAction().SYMBOL + p.getGroupAction().SYMBOL +
272                                         p.getOtherAction().SYMBOL;
273      }

275      /**
276       * Opens an FSDataOutputStream at the indicated Path with write-progress
277       * reporting.
278       * <p/>
279       * IMPORTANT: The <code>Progressable</code> parameter is not used.
280       *
281       * @param f the file name to open
282       * @param permission
283       * @param overwrite if a file with this name already exists, then if true,
284       *   the file will be overwritten, and if false an error will be thrown.
285       * @param bufferSize the size of the buffer to be used.
286       * @param replication required block replication for the file.
287       * @param blockSize
288       * @param progress
289       * @throws IOException
290       * @see #setPermission(Path, FsPermission)
291       */  @Override
292      public FSDataOutputStream create(Path f, FsPermission permission, boolean overwrite, int bufferSize,
293                                       short replication, long blockSize, Progressable progress) throws IOException {
294        Map<String, String> params = new HashMap<String, String>();
295        params.put("op", "create");
296        params.put("overwrite", Boolean.toString(overwrite));
297        params.put("replication", Short.toString(replication));
298        params.put("blocksize", Long.toString(blockSize));
299        params.put("permission", permissionToString(permission));
300        HttpURLConnection conn = getConnection("POST", params, f);
301        try {
302          OutputStream os = new BufferedOutputStream(conn.getOutputStream(), bufferSize);
303          return new HoopFSDataOutputStream(conn, os, HttpURLConnection.HTTP_CREATED, statistics);
304        }
305        catch (IOException ex) {
306          validateResponse(conn, HttpURLConnection.HTTP_CREATED);
307          throw ex;
308        }
309      }

382       * List the statuses of the files/directories in the given path if the path is
383       * a directory.
384       *
385       * @param f
386       *          given path
387       * @return the statuses of the files/directories in the given patch
388       * @throws IOException
389       */
390      @Override
391      public FileStatus[] listStatus(Path f) throws IOException {
392        Map<String, String> params = new HashMap<String, String>();
393        params.put("op", "list");
394        HttpURLConnection conn = getConnection("GET", params, f);
395        validateResponse(conn, HttpURLConnection.HTTP_OK);
396        JSONArray json = (JSONArray) jsonParse(conn);
397        FileStatus[] array = new FileStatus[json.size()];
398        for (int i = 0; i < json.size(); i++) {
399          array[i] = createFileStatus((JSONObject) json.get(i));
400        }
401        return array;
402      }

554      /**
555       * Creates a <code>FileStatus</code> object using a JSON file-status payload
556       * received from a Hoop server.
557       *
558       * @param json a JSON file-status payload received from a Hoop server
559       * @return the corresponding <code>FileStatus</code>
560       */
561      private FileStatus createFileStatus(JSONObject json) {
562        Path path = new Path((String) json.get("path"));
563        boolean isDir = (Boolean) json.get("isDir");
564        long len = (Long) json.get("len");
565        String owner = (String) json.get("owner");
566        String group = (String) json.get("group");
567        FsPermission permission = FsPermission.valueOf((String) json.get("permission"));
568        long aTime = (Long) json.get("accessTime");
569        long mTime = (Long) json.get("modificationTime");
570        long blockSize = (Long) json.get("blockSize");
571        short replication = (short) (long) (Long) json.get("replication");
572        return new FileStatus(len, isDir, replication, blockSize, mTime, aTime, permission, owner, group, path);
573      }
