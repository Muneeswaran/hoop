045      public enum Status {
046        UNDEF(false, false),
047        BOOTING(false, true),
048        HALTED(true, true),
049        ADMIN(true, true),
050        NORMAL(true, true),
051        SHUTTING_DOWN(false, true),
052        SHUTDOWN(false, false);
053    
054        private boolean settable;
055        private boolean operational;
056    
057        Status(boolean settable, boolean operational) {
058          this.settable = settable;
059          this.operational = operational;
060        }
061    
062        public boolean isOperational() {
063          return operational;
064        }
065      }

090      public Server(String name, String homeDir, String configDir, String logDir, String tempDir, XConfiguration config) {
091        this.name = Check.notEmpty(name, "name").trim().toLowerCase();
092        this.homeDir = Check.notEmpty(homeDir, "homeDir");
093        this.configDir = Check.notEmpty(configDir, "configDir");
094        this.logDir = Check.notEmpty(logDir, "logDir");
095        this.tempDir = Check.notEmpty(tempDir, "tempDir");
096        checkAbsolutePath(homeDir, "homeDir");
097        checkAbsolutePath(configDir, "configDir");
098        checkAbsolutePath(logDir, "logDir");
099        checkAbsolutePath(tempDir, "tempDir");
100        if (config != null) {
101          this.config = new XConfiguration();
102          XConfiguration.copy(config, this.config);
103        }
104        status = Status.UNDEF;
105      }
106    
107      private String checkAbsolutePath(String value, String name) {
108        if (!value.startsWith("/")) {
109          throw new IllegalArgumentException(
110            MessageFormat.format("[{0}] must be an absolute path [{1}]", name, value));
111        }
112        return value;
113      }

115      public Status getStatus() {
116        return status;
117      }

119      public void setStatus(Status status) throws ServerException {
120        Check.notNull(status, "status");
121        if (status.settable) {
122          if (status != this.status) {
123            Status oldStatus = this.status;
124            this.status = status;
125            for (Service service : services.values()) {
126              try {
127                service.serverStatusChange(oldStatus, status);
128              }
129              catch (Exception ex) {
130                log.error("Service [{}] exception during status change to [{}] -server shutting down-,  {}",
131                          new Object[]{service.getInterface().getSimpleName(), status, ex.getMessage(), ex});
132                destroy();
133                throw new ServerException(ServerException.ERROR.S11, service.getInterface().getSimpleName(),
134                                          status, ex.getMessage(), ex);
135              }
136            }
137          }
138        }
139        else {
140          throw new IllegalArgumentException("Status [" + status + " is not settable");
141        }
142      }

144      protected void ensureOperational() {
145        if (!getStatus().isOperational()) {
146          throw new IllegalStateException("Server is not running");
147        }
148      }

150      public void init() throws ServerException {
151        if (status != Status.UNDEF) {
152          throw new IllegalStateException("Server already initialized");
153        }
154        status = Status.BOOTING;
155        verifyDir(homeDir);
156        verifyDir(tempDir);
157        Properties serverInfo = new Properties();
158        try {
159          InputStream is = ClassUtils.getResource(name + ".properties");
160          serverInfo.load(is);
161          is.close();
162        }
163        catch (IOException ex) {
164          throw new RuntimeException("Could not load server information file: " + name + ".properties");
165        }
166        initLog();
167        log.info("++++++++++++++++++++++++++++++++++++++++++++++++++++++");
168        log.info("Server [{}] starting", name);
169        log.info("  Built information:");
170        log.info("    Version           : {}", serverInfo.getProperty(name + ".version", "undef"));
171        log.info("    Source Repository : {}", serverInfo.getProperty(name + ".source.repository", "undef"));
172        log.info("    Source Revision   : {}", serverInfo.getProperty(name + ".source.revision", "undef"));
173        log.info("    Built by          : {}", serverInfo.getProperty(name + ".build.username", "undef"));
174        log.info("    Built timestamp   : {}", serverInfo.getProperty(name + ".build.timestamp", "undef"));
175        log.info("  Runtime information:");
176        log.info("    Home   dir: {}", homeDir);
177        log.info("    Config dir: {}", (config == null) ? configDir : "-");
178        log.info("    Log    dir: {}", logDir);
179        log.info("    Temp   dir: {}", tempDir);
180        initConfig();
181        log.debug("Loading services");
182        List<Service> list = loadServices();
183        try {
184          log.debug("Initializing services");
185          initServices(list);
186          log.info("Services initialized");
187        }
188        catch (ServerException ex) {
189          log.error("Services initialization failure, destroying initialized services");
190          destroyServices();
191          throw ex;
192        }
193        Status status = Status.valueOf(getConfig().get(getPrefixedName(CONF_STARTUP_STATUS), Status.NORMAL.toString()));
194        setStatus(status);
195        log.info("Server [{}] started!, status [{}]", name, status);
196      }

198      private void verifyDir(String dir) throws ServerException {
199        File file = new File(dir);
200        if (!file.exists()) {
201          throw new ServerException(ServerException.ERROR.S01, dir);
202        }
203        if (!file.isDirectory()) {
204          throw new ServerException(ServerException.ERROR.S02, dir);
205        }
206      }
207    
208      protected void initLog() throws ServerException {
209        verifyDir(logDir);
210        LogManager.resetConfiguration();
211        File log4jFile = new File(configDir, name + "-log4j.properties");
212        if (log4jFile.exists()) {
213          PropertyConfigurator.configureAndWatch(log4jFile.toString(), 10 * 1000); //every 10 secs
214          log = LoggerFactory.getLogger(Server.class);
215        }
216        else {
217          Properties props = new Properties();
218          try {
219            InputStream is = ClassUtils.getResource(DEFAULT_LOG4J_PROPERTIES);
220            props.load(is);
221          }
222          catch (IOException ex) {
223            throw new ServerException(ServerException.ERROR.S03, DEFAULT_LOG4J_PROPERTIES, ex.getMessage(), ex);
224          }
225          PropertyConfigurator.configure(props);
226          log = LoggerFactory.getLogger(Server.class);
227          log.warn("Log4j [{}] configuration file not found, using default configuration from classpath", log4jFile);
228        }
229      }

231      protected void initConfig() throws ServerException {
232        verifyDir(configDir);
233        File file = new File(configDir);
234        Configuration defaultConf;
235        String defaultConfig = name + "-default.xml";
236        ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
237        InputStream inputStream = classLoader.getResourceAsStream(defaultConfig);
238        if (inputStream == null) {
239          log.warn("Default configuration file not available in classpath [{}]", defaultConfig);
240          defaultConf = new XConfiguration();
241        }
242        else {
243          try {
244            defaultConf = new XConfiguration(inputStream);
245          }
246          catch (IOException ex) {
247            throw new ServerException(ServerException.ERROR.S03, defaultConfig, ex.getMessage(), ex);
248          }
249        }
250    
251        if (config == null) {
252          XConfiguration siteConf;
253          File siteFile = new File(file, name + "-site.xml");
254          if (!siteFile.exists()) {
255            log.warn("Site configuration file [{}] not found in config directory", siteFile);
256            siteConf = new XConfiguration();
257          }
258          else {
259            if (!siteFile.isFile()) {
260              throw new ServerException(ServerException.ERROR.S05, siteFile.getAbsolutePath());
261            }
262            try {
263              log.debug("Loading site configuration from [{}]", siteFile);
264              inputStream = new FileInputStream(siteFile);
265              siteConf = new XConfiguration(inputStream);
266            }
267            catch (IOException ex) {
268              throw new ServerException(ServerException.ERROR.S06, siteFile, ex.getMessage(), ex);
269            }
270          }
271    
272          config = new XConfiguration();
273          XConfiguration.copy(siteConf, config);
274        }
275    
276        XConfiguration.injectDefaults(defaultConf, config);
277    
278        for (String name : System.getProperties().stringPropertyNames()) {
279          String value = System.getProperty(name);
280          if (name.startsWith(getPrefix() + ".")) {
281            config.set(name, value);
282            if (name.endsWith(".password") || name.endsWith(".secret")) {
283              value = "*MASKED*";
284            }
285            log.info("System property sets  {}: {}", name, value);
286          }
287        }
288    
289        log.debug("Loaded Configuration:");
290        log.debug("------------------------------------------------------");
291        for (Map.Entry<String, String> entry : config) {
292          String name = entry.getKey();
293          String value = config.get(entry.getKey());
294          if (name.endsWith(".password") || name.endsWith(".secret")) {
295            value = "*MASKED*";
296          }
297          log.debug("  {}: {}", entry.getKey(), value);
298        }
299        log.debug("------------------------------------------------------");
300      }

302      private void loadServices(Class[] classes, List<Service> list) throws ServerException {
303        for (Class klass : classes) {
304          try {
305            Service service = (Service) klass.newInstance();
306            log.debug("Loading service [{}] implementation [{}]", service.getInterface(),
307                      service.getClass());
308            if (!service.getInterface().isInstance(service)) {
309              throw new ServerException(ServerException.ERROR.S04, klass, service.getInterface().getName());
310            }
311            list.add(service);
312          }
313          catch (ServerException ex) {
314            throw ex;
315          }
316          catch (Exception ex) {
317            throw new ServerException(ServerException.ERROR.S07, klass, ex.getMessage(), ex);
318          }
319        }
320      }
321    
322      protected List<Service> loadServices() throws ServerException {
323        try {
324          Map<Class, Service> map = new LinkedHashMap<Class, Service>();
325          Class[] classes = getConfig().getClasses(getPrefixedName(CONF_SERVICES));
326          Class[] classesExt = getConfig().getClasses(getPrefixedName(CONF_SERVICES_EXT));
327          List<Service> list = new ArrayList<Service>();
328          loadServices(classes, list);
329          loadServices(classesExt, list);
330    
331          //removing duplicate services, strategy: last one wins
332          for (Service service : list) {
333            if (map.containsKey(service.getInterface())) {
334              log.debug("Replacing service [{}] implementation [{}]", service.getInterface(),
335                        service.getClass());
336            }
337            map.put(service.getInterface(), service);
338          }
339          list = new ArrayList<Service>();
340          for (Map.Entry<Class, Service> entry : map.entrySet()) {
341            list.add(entry.getValue());
342          }
343          return list;
344        }
345        catch (RuntimeException ex) {
346          throw new ServerException(ServerException.ERROR.S08, ex.getMessage(), ex);
347        }
348      }
349    
350      protected void initServices(List<Service> services) throws ServerException {
351        for (Service service : services) {
352          log.debug("Initializing service [{}]", service.getInterface());
353          checkServiceDependencies(service);
354          service.init(this);
355          this.services.put(service.getInterface(), service);
356        }
357        for (Service service : services) {
358          service.postInit();
359        }
360      }
361    
362      protected void checkServiceDependencies(Service service) throws ServerException {
363        if (service.getServiceDependencies() != null) {
364          for (Class dependency : service.getServiceDependencies()) {
365            if (services.get(dependency) == null) {
366              throw new ServerException(ServerException.ERROR.S10, service.getClass(), dependency);
367            }
368          }
369        }
370      }
371    
372      protected void destroyServices() {
373        List<Service> list = new ArrayList<Service>(services.values());
374        Collections.reverse(list);
375        for (Service service : list) {
376          try {
377            log.debug("Destroying service [{}]", service.getInterface());
378            service.destroy();
379          }
380          catch (Throwable ex) {
381            log.error("Could not destroy service [{}], {}",
382                      new Object[]{service.getInterface(), ex.getMessage(), ex});
383          }
384        }
385        log.info("Services destroyed");
386      }
387    
388      public void destroy() {
389        ensureOperational();
390        destroyServices();
391        log.info("Server [{}] shutdown!", name);
392        log.info("======================================================");
393        if (!Boolean.getBoolean("test.circus")) {
394          LogManager.shutdown();
395        }
396        status = Status.SHUTDOWN;
397      }

439      public void setService(Class<? extends Service> klass) throws ServerException {
440        ensureOperational();
441        Check.notNull(klass, "serviceKlass");
442        if (getStatus() == Status.SHUTTING_DOWN) {
443          throw new IllegalStateException("Server shutting down");
444        }
445        try {
446          Service newService = klass.newInstance();
447          Service oldService = services.get(newService.getInterface());
448          if (oldService != null) {
449            try {
450              oldService.destroy();
451            }
452            catch (Throwable ex) {
453              log.error("Could not destroy service [{}], {}",
454                        new Object[]{oldService.getInterface(), ex.getMessage(), ex});
455            }
456          }
457          newService.init(this);
458          services.put(newService.getInterface(), newService);
459        }
460        catch (Exception ex) {
461          log.error("Could not set service [{}] programmatically -server shutting down-, {}", klass, ex);
462          destroy();
463          throw new ServerException(ServerException.ERROR.S09, klass, ex.getMessage(), ex);
464        }
465      }
