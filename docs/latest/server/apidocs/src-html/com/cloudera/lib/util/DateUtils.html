015    package com.cloudera.lib.util;
016    
017    import java.sql.Timestamp;
018    import java.text.DateFormat;
019    import java.text.ParseException;
020    import java.text.SimpleDateFormat;
021    import java.util.Calendar;
022    import java.util.Date;
023    import java.util.GregorianCalendar;
024    import java.util.Locale;
025    import java.util.TimeZone;
026    
027    public class DateUtils {
028    
029      private static final TimeZone GMT = getTimeZone("GMT");
030    
031      public static String toRFC822(Date date) {
032        String str = null;
033        if (date != null) {
034          SimpleDateFormat dateFormater = new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss 'GMT'", Locale.US);
035          dateFormater.setTimeZone(GMT);
036          str = dateFormater.format(date);
037        }
038        return str;
039      }
040    
041      public static Date parseRFC822(String str) throws ParseException {
042        Date date = null;
043        if (str != null) {
044          SimpleDateFormat dateFormater = new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss 'GMT'", Locale.US);
045          dateFormater.setTimeZone(GMT);
046          date = dateFormater.parse(str);
047        }
048        return date;
049      }
050    
051      private static final TimeZone UTC = getTimeZone("UTC");
052    
053      private static DateFormat getISO8601DateFormat() {
054        DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm'Z'");
055        dateFormat.setTimeZone(UTC);
056        return dateFormat;
057      }
058    
059      public static TimeZone getTimeZone(String tzId) {
060        Check.notEmpty(tzId, "tzId");
061        TimeZone tz = TimeZone.getTimeZone(tzId);
062        if (!tz.getID().equals(tzId)) {
063          throw new IllegalArgumentException("Invalid TimeZone: " + tzId);
064        }
065        return tz;
066      }
067    
068      public static Date parseUTC(String str) throws ParseException {
069        Date date = null;
070        if (str != null) {
071          return getISO8601DateFormat().parse(str);
072        }
073        return date;
074      }
075    
076      public static String toUTC(Date date) throws Exception {
077        String str = null;
078        if (date != null) {
079          str = getISO8601DateFormat().format(date);
080        }
081        return str;
082      }
083    
084      public static String toUTC(Calendar calendar) throws Exception {
085        String str = null;
086        if (calendar != null) {
087          str = toUTC(calendar.getTime());
088        }
089        return str;
090      }
091    
092      public static int getHoursInDay(Calendar calendar) {
093        Check.notNull(calendar, "calendar");
094        Calendar localCal = new GregorianCalendar(calendar.getTimeZone());
095        localCal.set(Calendar.MILLISECOND, 0);
096        localCal.set(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH),
097                     calendar.get(Calendar.DAY_OF_MONTH), 0, 30, 0);
098        localCal.add(Calendar.HOUR_OF_DAY, 24);
099        switch (localCal.get(Calendar.HOUR_OF_DAY)) {
100          case 1:
101            return 23;
102          case 23:
103            return 25;
104          case 0:
105          default:
106            return 24;
107        }
108      }
109    
110      public static boolean isDSTChangeDay(Calendar calendar) {
111        return getHoursInDay(calendar) != 24;
112      }
113    
114      public static Calendar getCalendar(String strDate, TimeZone tz) throws Exception {
115        Check.notEmpty(strDate, "strDate");
116        Date date = DateUtils.parseUTC(strDate);
117        Calendar calDate = Calendar.getInstance();
118        calDate.setTime(date);
119        calDate.setTimeZone(tz);
120        return calDate;
121      }
122    
123      public static Calendar getCalendar(String strDate) throws Exception {
124        return getCalendar(strDate, DateUtils.getTimeZone("UTC"));
125      }
126    
127      public static Date toDate(Timestamp timestamp) {
128        Date date = null;
129        if (timestamp != null) {
130          date = new Date(timestamp.getTime());
131        }
132        return date;
133      }
134    
135      public static Timestamp toTimestamp(Date date) {
136        Timestamp ts = null;
137        if (date != null) {
138          ts = new Timestamp(date.getTime());
139        }
140        return ts;
141      }
142    
143    }
