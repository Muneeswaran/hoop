015    package com.cloudera.lib.servlet;
016    
017    import com.cloudera.lib.service.Hadoop;
018    import org.apache.hadoop.fs.FileSystem;
019    
020    import javax.servlet.Filter;
021    import javax.servlet.FilterChain;
022    import javax.servlet.FilterConfig;
023    import javax.servlet.ServletException;
024    import javax.servlet.ServletRequest;
025    import javax.servlet.ServletResponse;
026    import java.io.IOException;
027    
028    /**
029     * The <code>FileSystemReleaseFilter</code> releases back to the
030     * {@link Hadoop} service a <code>FileSystem</code> instance.
031     * <p/>
032     * This filter is useful in situations where a servlet request
033     * is streaming out HDFS data and the corresponding filesystem
034     * instance have to be closed after the streaming completes.
035     */
036    public abstract class FileSystemReleaseFilter implements Filter {
037      private static final ThreadLocal<FileSystem> FILE_SYSTEM_TL = new ThreadLocal<FileSystem>();
038    
039      /**
040       * Initializes the filter.
041       * <p/>
042       * This implementation is a NOP.
043       *
044       * @param filterConfig filter configuration.
045       * @throws ServletException thrown if the filter could not be initialized.
046       */
047      @Override
048      public void init(FilterConfig filterConfig) throws ServletException {
049      }
050    
051      /**
052       * It delegates the incoming request to the <code>FilterChain</code>, and
053       * at its completion (in a finally block) releases the filesystem instance
054       * back to the {@link Hadoop} service.
055       *
056       * @param servletRequest servlet request.
057       * @param servletResponse servlet response.
058       * @param filterChain filter chain.
059       * @throws IOException thrown if an IO error occurrs.
060       * @throws ServletException thrown if a servet error occurrs.
061       */
062      @Override
063      public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain)
064        throws IOException, ServletException {
065        try {
066          filterChain.doFilter(servletRequest, servletResponse);
067        }
068        finally {
069          FileSystem fs = FILE_SYSTEM_TL.get();
070          if (fs != null) {
071            FILE_SYSTEM_TL.remove();
072            getHadoop().releaseFileSystem(fs);
073          }
074        }
075      }
076    
077      /**
078       * Destroys the filter.
079       * <p/>
080       * This implementation is a NOP.
081       */
082      @Override
083      public void destroy() {
084      }
085    
086      /**
087       * Static method that sets the <code>FileSystem</code> to release back to
088       * the {@link Hadoop} service on servlet request completion.
089       *
090       * @param fs fileystem instance.
091       */
092      public static void setFileSystem(FileSystem fs) {
093        FILE_SYSTEM_TL.set(fs);
094      }
095    
096      /**
097       * Abstract method to be implemetned by concrete implementations of the
098       * filter that return the {@link Hadoop} service to which the filesystem
099       * will be returned to.
100       *
101       * @return the Hadoop service.
102       */
103      protected abstract Hadoop getHadoop();
104    
105    }
