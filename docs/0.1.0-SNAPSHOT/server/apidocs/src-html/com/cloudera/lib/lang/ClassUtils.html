038      public static String getJar(Class klass) {
039        Check.notNull(klass, "klass");
040        ClassLoader loader = klass.getClassLoader();
041        if (loader != null) {
042          String class_file = klass.getName().replaceAll("\\.", "/") + ".class";
043          try {
044            for (Enumeration itr = loader.getResources(class_file); itr.hasMoreElements(); ) {
045              URL url = (URL) itr.nextElement();
046              if ("jar".equals(url.getProtocol())) {
047                String toReturn = url.getPath();
048                if (toReturn.startsWith("file:")) {
049                  toReturn = toReturn.substring("file:".length());
050                }
051                toReturn = URLDecoder.decode(toReturn, "UTF-8");
052                return toReturn.replaceAll("!.*$", "");
053              }
054            }
055          }
056          catch (IOException e) {
057            throw new RuntimeException(e);
058          }
059        }
060        return null;
061      }
062    
063      public static InputStream getResource(String name) {
064        Check.notEmpty(name, "name");
065        ClassLoader cl = Thread.currentThread().getContextClassLoader();
066        if (cl == null) {
067          cl = ClassUtils.class.getClassLoader();
068        }
069        return cl.getResourceAsStream(name);
070      }
071    
072      public static void createJar(File jarFile, Class... classes) throws IOException {
073        Check.notNull(jarFile, "jarFile");
074        File jarDir = jarFile.getParentFile();
075        if (!jarDir.exists()) {
076          if (!jarDir.mkdirs()) {
077            throw new IOException(MessageFormat.format("could not create dir [{0}]", jarDir));
078          }
079        }
080        createJar(new FileOutputStream(jarFile), classes);
081      }
082    
083      public static void createJar(OutputStream os, Class... classes) throws IOException {
084        Check.notNull(os, "os");
085        File classesDir = File.createTempFile("createJar", "classes");
086        if (!classesDir.delete()) {
087          throw new IOException(MessageFormat.format("could not delete temp file [{0}]", classesDir));
088        }
089        for (Class clazz : classes) {
090          String classPath = clazz.getName().replace(".", "/") + ".class";
091          String classFileName = classPath;
092          if (classPath.lastIndexOf("/") > -1) {
093            classFileName = classPath.substring(classPath.lastIndexOf("/") + 1);
094          }
095          String packagePath = new File(classPath).getParent();
096          File dir = new File(classesDir, packagePath);
097          if (!dir.exists()) {
098            if (!dir.mkdirs()) {
099              throw new IOException(MessageFormat.format("could not create dir [{0}]", dir));
100            }
101          }
102          InputStream is = getResource(classPath);
103          OutputStream classOS = new FileOutputStream(new File(dir, classFileName));
104          IOUtils.copy(is, classOS);
105        }
106        JarOutputStream zos = new JarOutputStream(os, new Manifest());
107        IOUtils.zipDir(classesDir, "", zos);
108        IOUtils.delete(classesDir);
109      }

111      public static Method findMethod(String className, String methodName) throws ServiceException {
112        Check.notEmpty(className, "className");
113        Check.notEmpty(methodName, "methodName");
114        Method method = null;
115        try {
116          Class klass = Thread.currentThread().getContextClassLoader().loadClass(className);
117          for (Method m : klass.getMethods()) {
118            if (m.getName().equals(methodName)) {
119              method = m;
120              break;
121            }
122          }
123          if (method == null) {
124            throw new IllegalArgumentException(MessageFormat.format("class#method not found [{0}#{1}]", className,
125                                                                    methodName));
126          }
127          if ((method.getModifiers() & (Modifier.PUBLIC | Modifier.STATIC)) != (Modifier.PUBLIC | Modifier.STATIC)) {
128            throw new IllegalArgumentException(MessageFormat.format(
129              "class#method does not have PUBLIC or STATIC modifier [{0}#{1}]", className, methodName));
130          }
131        }
132        catch (ClassNotFoundException ex) {
133          throw new IllegalArgumentException(MessageFormat.format("class not found [{0}]", className));
134        }
135        return method;
136      }
137    
138      public static Object findConstant(String className, String constantName) throws ServiceException {
139        Check.notEmpty(className, "className");
140        Check.notEmpty(constantName, "constantName");
141        try {
142          Class klass = Thread.currentThread().getContextClassLoader().loadClass(className);
143          Field field = klass.getField(constantName);
144          if ((field.getModifiers() & (Modifier.PUBLIC | Modifier.STATIC)) != (Modifier.PUBLIC | Modifier.STATIC)) {
145            throw new IllegalArgumentException(MessageFormat.format(
146              "class#constant does not have PUBLIC or STATIC modifier [{0}#{1}]", className, constantName));
147          }
148          return field.get(null);
149        }
150        catch (IllegalAccessException ex) {
151          throw new IllegalArgumentException(ex);
152        }
153        catch (NoSuchFieldException ex) {
154          throw new IllegalArgumentException(MessageFormat.format("class#constant not found [{0}#{1}]", className,
155                                                                  constantName));
156        }
157        catch (ClassNotFoundException ex) {
158          throw new IllegalArgumentException(MessageFormat.format("class not found [{0}]", className));
159        }
160      }
