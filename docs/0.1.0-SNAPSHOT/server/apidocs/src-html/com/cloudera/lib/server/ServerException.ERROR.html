015    package com.cloudera.lib.server;
016    
017    import com.cloudera.lib.lang.XException;
018    
019    public class ServerException extends XException {
020    
021      public static enum ERROR implements XException.ERROR {
022        S01("Dir [{0}] does not exist"),
023        S02("[{0}] is not a directory"),
024        S03("Could not load file from classpath [{0}], {1}"),
025        S04("Service [{0}] does not implement declared interface [{1}]"),
026        S05("[{0}] is not a file"),
027        S06("Could not load file [{0}], {1}"),
028        S07("Could not instanciate service class [{0}], {1}"),
029        S08("Could not load service classes, {0}"),
030        S09("Could not set service [{0}] programmatically -server shutting down-, {1}"),
031        S10("Service [{0}] requires service [{1}]"),
032        S11("Service [{0}] exception during status change to [{1}] -server shutting down-, {2}");
033    
034        private String msg;
035    
036        private ERROR(String msg) {
037          this.msg = msg;
038        }
039    
040        @Override
041        public String getTemplate() {
042          return msg;
043        }
044      }
045    
046      protected ServerException(XException.ERROR error, Object... params) {
047        super(error, params);
048      }
049    
050      public ServerException(ERROR error, Object... params) {
051        super(error, params);
052      }
053    
054    }
