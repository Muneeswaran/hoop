015    package com.cloudera.hoop;
016    
017    import com.cloudera.alfredo.server.AuthenticationFilter;
018    import org.apache.hadoop.conf.Configuration;
019    
020    import javax.servlet.FilterConfig;
021    import java.util.Map;
022    import java.util.Properties;
023    
024    public class AuthFilter extends AuthenticationFilter {
025      private static final String CONF_PREFIX = "hoop.authentication.";
026    
027      @Override
028      protected Properties getConfiguration(String configPrefix, FilterConfig filterConfig) {
029        Properties props = new Properties();
030        Configuration conf = HoopServer.get().getConfig();
031    
032        props.setProperty(AuthenticationFilter.COOKIE_PATH, "/");
033        for (Map.Entry<String, String> entry : conf) {
034          String name = entry.getKey();
035          if (name.startsWith(CONF_PREFIX)) {
036            String value = conf.get(name);
037            name = name.substring(CONF_PREFIX.length());
038            props.setProperty(name, value);
039          }
040        }
041        return props;
042      }
043    
044    
045    }
