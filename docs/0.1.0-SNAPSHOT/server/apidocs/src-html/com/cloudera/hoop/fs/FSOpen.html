015    package com.cloudera.hoop.fs;
016    
017    import com.cloudera.hoop.HoopServer;
018    import com.cloudera.lib.service.Hadoop;
019    import org.apache.hadoop.fs.FileSystem;
020    import org.apache.hadoop.fs.Path;
021    
022    import java.io.IOException;
023    import java.io.InputStream;
024    
025    /**
026     * Executor that performs a open Hadoop files system operation.
027     */
028    public class FSOpen implements Hadoop.FileSystemExecutor<InputStream> {
029      private Path path;
030    
031      /**
032       * Creates a open executor.
033       *
034       * @param path file to open.
035       */
036      public FSOpen(String path) {
037        this.path = new Path(path);
038      }
039    
040      /**
041       * Executes the filesystem operation.
042       *
043       * @param fs filesystem instance to use.
044       * @return The inputstream of the file.
045       * @throws IOException thrown if an IO error occured.
046       */
047      @Override
048      public InputStream execute(FileSystem fs) throws IOException {
049        int bufferSize = HoopServer.get().getConfig().getInt("hoop.buffer.size", 4096);
050        return fs.open(path, bufferSize);
051      }
052    
053    }
