120      /**
121       * Resolves the effective user that will be used to request a Hadoop filesystem.
122       * <p/>
123       * If the doAs-user is NULL or the same as the user, it returns the user.
124       * <p/>
125       * Otherwise it uses proxyuser rules (see {@link ProxyUser} to determine if the
126       * current user can impersonate the doAs-user.
127       * <p/>
128       * If the current user cannot impersonate the doAs-user an
129       * <code>AccessControlException</code> will be thrown.
130       *
131       * @param user principal for whom the filesystem instance is.
132       * @param doAs do-as user, if any.
133       * @return the effective user.
134       * @throws IOException thrown if an IO error occurrs.
135       * @throws AccessControlException thrown if the current user cannot impersonate
136       * the doAs-user.
137       */
138      private String getEffectiveUser(Principal user, String doAs) throws IOException {
139        String effectiveUser = user.getName();
140        if (doAs != null && !doAs.equals(user.getName())) {
141          ProxyUser proxyUser = HoopServer.get().get(ProxyUser.class);
142          proxyUser.validate(user.getName(), HostnameFilter.get(), doAs);
143          effectiveUser = doAs;
144          AUDIT_LOG.info("Proxy user [{}] DoAs user [{}]", user.getName(), doAs);
145        }
146        return effectiveUser;
147      }

149      /**
150       * Executes a {@link Hadoop.FileSystemExecutor} using a filesystem for the effective
151       * user.
152       * @param user principal making the request.
153       * @param doAs do-as user, if any.
154       * @param executor FileSystemExecutor to execute.
155       * @return  FileSystemExecutor response
156       * @throws IOException thrown if an IO error occurrs.
157       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
158       * exceptions are handled by {@link HoopExceptionProvider}.
159       */
160      private <T> T fsExecute(Principal user, String doAs, Hadoop.FileSystemExecutor<T> executor)
161        throws IOException, HadoopException {
162        String hadoopUser = getEffectiveUser(user, doAs);
163        Hadoop hadoop = HoopServer.get().get(Hadoop.class);
164        Configuration conf = HoopServer.get().get(Hadoop.class).getDefaultConfiguration();
165        return hadoop.execute(hadoopUser, conf, executor);
166      }

168      /**
169       * Returns a filesystem instance. The fileystem instance is wired for release at the completion of
170       * the current Servlet request via the {@link FileSystemReleaseFilter}.
171       * <p/>
172       * If a do-as user is specified, the current user must be a valid proxyuser, otherwise an
173       * <code>AccessControlException</code> will be thrown.
174       *
175       * @param user principal for whom the filesystem instance is.
176       * @param doAs do-as user, if any.
177       * @return a filesystem for the specified user or do-as user.
178       * @throws IOException thrown if an IO error occurred. Thrown exceptions are
179       * handled by {@link HoopExceptionProvider}.
180       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
181       * exceptions are handled by {@link HoopExceptionProvider}.
182       */
183      private FileSystem createFileSystem(Principal user, String doAs) throws IOException, HadoopException {
184        String hadoopUser = getEffectiveUser(user, doAs);
185        Hadoop hadoop = HoopServer.get().get(Hadoop.class);
186        Configuration conf = HoopServer.get().get(Hadoop.class).getDefaultConfiguration();
187        FileSystem fs = hadoop.createFileSystem(hadoopUser, conf);
188        FileSystemReleaseFilter.setFileSystem(fs);
189        return fs;
190      }

192      /**
193       * Binding to handle all GET requests, supported operations are
194       * {@link GetOpParam.Values}.
195       * <p/>
196       * The {@link GetOpParam.Values#INSTRUMENTATION} operation is available only
197       * to users that are in Hoop's admin group (see {@link HoopServer}. It returns
198       * Hoop instrumentation data. The specified path must be '/'.
199       *
200       * @param user principal making the request.
201       * @param path path for the GET request.
202       * @param op GET operation, default value is {@link GetOpParam.Values#DATA}.
203       * @param offset of the  file being fetch, used only with
204       * {@link GetOpParam.Values#DATA} operations.
205       * @param len amounts of bytes, used only with {@link GetOpParam.Values#DATA}
206       * operations.
207       * @param filter Glob filter, default value is none. Used only if the
208       * operation is {@link GetOpParam.Values#LIST}
209       * @param doAs user being impersonated, defualt value is none. It can be used
210       * only if the current user is a Hoop proxyuser.
211       * @return the request response.
212       * @throws IOException thrown if an IO error occurred. Thrown exceptions are
213       * handled by {@link HoopExceptionProvider}.
214       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
215       * exceptions are handled by {@link HoopExceptionProvider}.
216       */
217      @GET
218      @Path("{path:.*}")
219      @Produces({MediaType.APPLICATION_OCTET_STREAM, MediaType.APPLICATION_JSON})
220      public Response get(@Context Principal user,
221                          @PathParam("path") @DefaultValue("") FsPathParam path,
222                          @QueryParam(GetOpParam.NAME) @DefaultValue(GetOpParam.DEFAULT) GetOpParam op,
223                          @QueryParam(OffsetParam.NAME) @DefaultValue(OffsetParam.DEFAULT) OffsetParam offset,
224                          @QueryParam(LenParam.NAME) @DefaultValue(LenParam.DEFAULT) LenParam len,
225                          @QueryParam(FilterParam.NAME) @DefaultValue(FilterParam.DEFAULT) FilterParam filter,
226                          @QueryParam(DoAsParam.NAME) @DefaultValue(DoAsParam.DEFAULT) DoAsParam doAs)
227        throws IOException, HadoopException {
228        Response response = null;
229        path.makeAbsolute();
230        MDC.put("op", op.value().name());
231        switch (op.value()) {
232          case DATA: {
233            //Invoking the command directly using an unmanaged FileSystem that is released by the
234            //FileSystemReleaseFilter
235            FSOpen command = new FSOpen(path.value());
236            FileSystem fs = createFileSystem(user, doAs.value());
237            InputStream is = command.execute(fs);
238            AUDIT_LOG.info("[{}] offset [{}] len [{}]", new Object[]{path, offset, len});
239            InputStreamEntity entity = new InputStreamEntity(is, offset.value(), len.value());
240            response = Response.ok(entity).type(MediaType.APPLICATION_OCTET_STREAM).build();
241            break;
242          }
243          case STATUS: {
244            FSFileStatus command = new FSFileStatus(path.value());
245            Map json = fsExecute(user, doAs.value(), command);
246            AUDIT_LOG.info("[{}]", path);
247            response = Response.ok(json).type(MediaType.APPLICATION_JSON).build();
248            break;
249          }
250          case LIST: {
251            FSListStatus command = new FSListStatus(path.value(), filter.value());
252            JSONArray json = fsExecute(user, doAs.value(), command);
253            if (filter.value() == null) {
254              AUDIT_LOG.info("[{}]", path);
255            }
256            else {
257              AUDIT_LOG.info("[{}] filter [{}]", path, filter.value());
258            }
259            response = Response.ok(json).type(MediaType.APPLICATION_JSON).build();
260            break;
261          }
262          case HOMEDIR: {
263            FSHomeDir command = new FSHomeDir();
264            JSONObject json = fsExecute(user, doAs.value(), command);
265            AUDIT_LOG.info("");
266            response = Response.ok(json).type(MediaType.APPLICATION_JSON).build();
267            break;
268          }
269          case INSTRUMENTATION: {
270            if (!path.value().equals("/")) {
271              throw new UnsupportedOperationException(
272                MessageFormat.format("Invalid path for {0}={1}, must be '/'",
273                                     GetOpParam.NAME, GetOpParam.Values.INSTRUMENTATION));
274            }
275            Groups groups = HoopServer.get().get(Groups.class);
276            List<String> userGroups = groups.getGroups(user.getName());
277            if (!userGroups.contains(HoopServer.get().getAdminGroup())) {
278              throw new AccessControlException("User not in Hoop admin group");
279            }
280            Instrumentation instrumentation = HoopServer.get().get(Instrumentation.class);
281            Map snapshot = instrumentation.getSnapshot();
282            response = Response.ok(snapshot).build();
283            break;
284          }
285        }
286        return response;
287      }

290       * Binding to handle all DELETE requests.
291       *
292       * @param user principal making the request.
293       * @param path path for the DELETE request.
294       * @param recursive indicates if the delete is recursive, default is <code>false</code>
295       * @param doAs user being impersonated, defualt value is none. It can be used
296       * only if the current user is a Hoop proxyuser.
297       * @return the request response.
298       * @throws IOException thrown if an IO error occurred. Thrown exceptions are
299       * handled by {@link HoopExceptionProvider}.
300       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
301       * exceptions are handled by {@link HoopExceptionProvider}.
302       */
303      @DELETE
304      @Path("{path:.*}")
305      @Produces(MediaType.APPLICATION_JSON)
306      public Response delete(@Context Principal user,
307                             @PathParam("path") FsPathParam path,
308                             @QueryParam(DeleteRecursiveParam.NAME) @DefaultValue(DeleteRecursiveParam.DEFAULT)
309                             DeleteRecursiveParam recursive,
310                             @QueryParam(DoAsParam.NAME) @DefaultValue(DoAsParam.DEFAULT) DoAsParam doAs)
311        throws IOException, HadoopException {
312        path.makeAbsolute();
313        MDC.put("op", "DELETE");
314        AUDIT_LOG.info("[{}] recursive [{}]", path, recursive);
315        FSDelete command = new FSDelete(path.value(), recursive.value());
316        JSONObject json = fsExecute(user, doAs.value(), command);
317        return Response.ok(json).type(MediaType.APPLICATION_JSON).build();
318      }

321      /**
322       * Binding to handle all PUT requests, supported operations are
323       * {@link PutOpParam.Values}.
324       *
325       * @param is request input stream, used only for
326       * {@link PutOpParam.Values#APPEND} operations.
327       * @param user principal making the request.
328       * @param path path for the PUT request.
329       * @param op PUT operation, no default value.
330       * @param toPath new path, used only for
331       * {@link PutOpParam.Values#RENAME} operations.
332       * {@link PutOpParam.Values#SETTIMES}.
333       * @param owner owner to set, used only for
334       * {@link PutOpParam.Values#SETOWNER} operations.
335       * @param group group to set, used only for
336       * {@link PutOpParam.Values#SETOWNER} operations.
337       * @param permission permission to set, used only by
338       * {@link PutOpParam.Values#SETPERMISSION}.
339       * @param replication replication factor to set, used only by
340       * {@link PutOpParam.Values#SETREPLICATION}.
341       * @param modifiedTime modified time, in seconds since EPOC, used only by
342       * {@link PutOpParam.Values#SETTIMES}.
343       * @param accessTime accessed time, in seconds since EPOC, used only by
344       * {@link PutOpParam.Values#SETTIMES}.
345       * @param doAs user being impersonated, defualt value is none. It can be used
346       * only if the current user is a Hoop proxyuser.
347       * @return the request response.
348       * @throws IOException thrown if an IO error occurred. Thrown exceptions are
349       * handled by {@link HoopExceptionProvider}.
350       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
351       * exceptions are handled by {@link HoopExceptionProvider}.
352       */
353      @PUT
354      @Path("{path:.*}")
355      @Consumes({"*/*"})
356      @Produces({MediaType.APPLICATION_JSON})
357      public Response put(InputStream is,
358                          @Context Principal user,
359                          @PathParam("path") FsPathParam path,
360                          @QueryParam(PutOpParam.NAME) PutOpParam op,
361                          @QueryParam(ToPathParam.NAME) @DefaultValue(ToPathParam.DEFAULT) ToPathParam toPath,
362                          @QueryParam(OwnerParam.NAME) @DefaultValue(OwnerParam.DEFAULT) OwnerParam owner,
363                          @QueryParam(GroupParam.NAME) @DefaultValue(GroupParam.DEFAULT) GroupParam group,
364                          @QueryParam(PermissionParam.NAME) @DefaultValue(PermissionParam.DEFAULT)
365                          PermissionParam permission,
366                          @QueryParam(ReplicationParam.NAME) @DefaultValue(ReplicationParam.DEFAULT)
367                          ReplicationParam replication,
368                          @QueryParam(ModifiedTimeParam.NAME) @DefaultValue(ModifiedTimeParam.DEFAULT)
369                          ModifiedTimeParam modifiedTime,
370                          @QueryParam(AccessTimeParam.NAME) @DefaultValue(AccessTimeParam.DEFAULT)
371                          AccessTimeParam accessTime,
372                          @QueryParam(DoAsParam.NAME) @DefaultValue(DoAsParam.DEFAULT) DoAsParam doAs)
373        throws IOException, HadoopException {
374        Response response = null;
375        path.makeAbsolute();
376        if (op == null) {
377          throw new UnsupportedOperationException(MessageFormat.format("Missing [{0}] parameter", PutOpParam.NAME));
378        }
379        MDC.put("op", op.value().name());
380        switch (op.value()) {
381          case APPEND: {
382            FSAppend command = new FSAppend(is, path.value());
383            fsExecute(user, doAs.value(), command);
384            AUDIT_LOG.info("[{}]", path);
385            response = Response.ok().type(MediaType.APPLICATION_JSON).build();
386            break;
387          }
388          case RENAME: {
389            FSRename command = new FSRename(path.value(), toPath.value());
390            JSONObject json = fsExecute(user, doAs.value(), command);
391            AUDIT_LOG.info("[{}] to [{}]", path, toPath);
392            response = Response.ok(json).type(MediaType.APPLICATION_JSON).build();
393            break;
394          }
395          case SETOWNER: {
396            FSSetOwner command = new FSSetOwner(path.value(), owner.value(), group.value());
397            fsExecute(user, doAs.value(), command);
398            AUDIT_LOG.info("[{}] to (O/G)[{}]", path, owner.value() + ":" + group.value());
399            response = Response.ok().build();
400            break;
401          }
402          case SETPERMISSION: {
403            FSSetPermission command = new FSSetPermission(path.value(), permission.value());
404            fsExecute(user, doAs.value(), command);
405            AUDIT_LOG.info("[{}] to [{}]", path, permission.value());
406            response = Response.ok().build();
407            break;
408          }
409          case SETREPLICATION: {
410            FSSetReplication command = new FSSetReplication(path.value(), replication.value());
411            JSONObject json = fsExecute(user, doAs.value(), command);
412            AUDIT_LOG.info("[{}] to [{}]", path, replication.value());
413            response = Response.ok(json).build();
414            break;
415          }
416          case SETTIMES: {
417            FSSetTimes command = new FSSetTimes(path.value(), modifiedTime.value(), accessTime.value());
418            fsExecute(user, doAs.value(), command);
419            AUDIT_LOG.info("[{}] to (M/A)[{}]", path, modifiedTime.value() + ":" + accessTime.value());
420            response = Response.ok().build();
421            break;
422          }
423        }
424        return response;
425      }

427      /**
428       * Binding to handle all OPST requests, supported operations are
429       * {@link PostOpParam.Values}.
430       *
431       * @param is request input stream, used only for
432       * {@link PostOpParam.Values#CREATE} operations.
433       * @param user principal making the request.
434       * @param path path for the POST request.
435       * @param op POST operation, default is {@link PostOpParam.Values#CREATE}.
436       * @param override, default is true. Used only for
437       * {@link PostOpParam.Values#CREATE} operations.
438       * @param replication replication factor to set, used only by
439       * {@link PostOpParam.Values#CREATE} operations.
440       * @param blockSize block size to set, used only by
441       * {@link PostOpParam.Values#CREATE} operations.
442       * @param permission permission to set.
443       * @param doAs user being impersonated, defualt value is none. It can be used
444       * only if the current user is a Hoop proxyuser.
445       * @return the request response.
446       * @throws IOException thrown if an IO error occurred. Thrown exceptions are
447       * handled by {@link HoopExceptionProvider}.
448       * @throws HadoopException thrwon if a Hadoop releated error occurred. Thrown
449       * exceptions are handled by {@link HoopExceptionProvider}.
450       */
451      @POST
452      @Path("{path:.*}")
453      @Consumes({"*/*"})
454      @Produces({MediaType.APPLICATION_JSON})
455      public Response post(InputStream is,
456                           @Context Principal user,
457                           @PathParam("path") FsPathParam path,
458                           @QueryParam(PostOpParam.NAME) @DefaultValue(PostOpParam.DEFAULT) PostOpParam op,
459                           @QueryParam(OverwriteParam.NAME) @DefaultValue(OverwriteParam.DEFAULT)
460                           OverwriteParam override,
461                           @QueryParam(ReplicationParam.NAME) @DefaultValue(ReplicationParam.DEFAULT)
462                           ReplicationParam replication,
463                           @QueryParam(BlockSizeParam.NAME) @DefaultValue(BlockSizeParam.DEFAULT)
464                           BlockSizeParam blockSize,
465                           @QueryParam(PermissionParam.NAME) @DefaultValue(PermissionParam.DEFAULT)
466                           PermissionParam permission,
467                           @QueryParam(DoAsParam.NAME) @DefaultValue(DoAsParam.DEFAULT) DoAsParam doAs)
468        throws IOException, HadoopException {
469        Response response = null;
470        path.makeAbsolute();
471        MDC.put("op", op.value().name());
472        switch (op.value()) {
473          case CREATE: {
474            FSCreate command = new FSCreate(is, path.value(), permission.value(), override.value(),
475                                            replication.value(), blockSize.value());
476            URI uri = fsExecute(user, doAs.value(), command);
477            AUDIT_LOG.info("[{}] permission [{}] override [{}] replication [{}] blockSize [{}]",
478                           new Object[]{path, permission, override, replication, blockSize});
479            response = Response.created(uri).type(MediaType.APPLICATION_JSON).build();
480            break;
481          }
482          case MKDIRS: {
483            FSMkdirs command = new FSMkdirs(path.value(), permission.value());
484            JSONObject json = fsExecute(user, doAs.value(), command);
485            AUDIT_LOG.info("[{}] permission [{}]", path, permission.value());
486            response = Response.ok(json).type(MediaType.APPLICATION_JSON).build();
487            break;
488          }
489        }
490        return response;
491      }
